364  virtual void construct(ASTStack &st)
365  {
366  for(;;)
367  {
368  //if the stack is empty

370 
371  //get the node
372  ASTNode *node = st.back().get();
373

376 
377  //if the object was not not of the appropriate type,
378  //end the list parsing
379  if (!obj) return;
380 
381  //remove the node from the stack
382  st.back().release();
383  st.pop_back();
384 
385  //insert the object in the list, in reverse order
386  child_objects.push_front(std::unique_ptr<T>(obj));
